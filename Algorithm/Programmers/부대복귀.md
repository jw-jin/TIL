from collections import deque

def bfs(graph, st):
    q = deque()
    visited = [0] * (ncnt+1)
    time = [0] * (ncnt+1)
    q.append(st)
    visited[st] = 1
    cnt = 0
    while q:
        nd = q.popleft()
        if nd in graph:
            for n in graph[nd]:
                if visited[n] == 0:
                    q.append(n)
                    visited[n] = 1
                    time[n] = time[nd] + 1
    
    return time
